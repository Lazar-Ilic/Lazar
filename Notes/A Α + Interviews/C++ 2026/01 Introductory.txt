#include <iostream>
#include <string>
#include <queue>
#include <unordered_map>
#include <vector>

using namespace std;

// A Huffman Tree Node
struct Node {
    char ch;
    int freq;
    Node *left, *right;

    // Constructor for a leaf node
    Node(char ch, int freq) {
        this->ch = ch;
        this->freq = freq;
        left = right = nullptr;
    }

    // Constructor for an internal node
    Node(char ch, int freq, Node* left, Node* right) {
        this->ch = ch;
        this->freq = freq;
        this->left = left;
        this->right = right;
    }
};

// Custom comparator to order the priority queue (Min-Heap)
struct compare {
    bool operator()(Node* l, Node* r) {
        // We want the node with the lowest frequency to have the highest priority
        return l->freq > r->freq;
    }
};

// Function to traverse the Huffman Tree and store the generated codes
void encode(Node* root, string str, unordered_map<char, string> &huffmanCode) {
    if (root == nullptr) {
        return;
    }

    // If it's a leaf node, it contains a character
    if (!root->left && !root->right) {
        huffmanCode[root->ch] = str;
    }

    // Traverse left (add '0') and right (add '1')
    encode(root->left, str + "0", huffmanCode);
    encode(root->right, str + "1", huffmanCode);
}

// Memory management helper to delete the tree
void freeTree(Node* root) {
    if (root == nullptr) return;
    freeTree(root->left);
    freeTree(root->right);
    delete root;
}

// Function to build the Huffman Tree and generate codes
void buildHuffmanTree(string text) {
    if (text.empty()) {
        return;
    }

    // 1. Count the frequency of each character
    unordered_map<char, int> freq;
    for (char ch : text) {
        freq[ch]++;
    }

    // 2. Create a priority queue to store live nodes of the Huffman tree
    priority_queue<Node*, vector<Node*>, compare> pq;

    // Create a leaf node for each character and add it to the priority queue
    for (auto pair : freq) {
        pq.push(new Node(pair.first, pair.second));
    }

    // 3. Build the Huffman Tree
    while (pq.size() != 1) {
        // Remove the two nodes of the lowest frequency
        Node *left = pq.top(); pq.pop();
        Node *right = pq.top(); pq.pop();

        // Create a new internal node with a frequency equal to the sum of the two nodes.
        // We use '\0' (null character) for internal nodes.
        int sum = left->freq + right->freq;
        pq.push(new Node('\0', sum, left, right));
    }

    // The remaining node is the root of the Huffman Tree
    Node* root = pq.top();

    // 4. Traverse the tree and generate the codes
    unordered_map<char, string> huffmanCode;
    encode(root, "", huffmanCode);

    // Print the generated Huffman codes
    cout << "Huffman Codes are:\n";
    for (auto pair : huffmanCode) {
        cout << pair.first << " : " << pair.second << '\n';
    }

    // Print the original string and the encoded string
    cout << "\nOriginal string:\n" << text << '\n';
    
    string encodedString = "";
    for (char ch : text) {
        encodedString += huffmanCode[ch];
    }
    cout << "\nEncoded string:\n" << encodedString << '\n';

    // Clean up dynamically allocated memory
    freeTree(root);
}

int main() {
    string text = "C++ Huffman Coding Example";
    
    buildHuffmanTree(text);
    
    return 0;
}

int bitap_search(std::string_view stream, std::string_view target) {
    const int m = (int)target.size();
    if (m == 0) return 0;
    if (m > 64) return -1; // use a different algorithm for longer patterns

    // pattern_mask[c]: bit i is SET if target[i] != c, CLEAR if target[i] == c
    uint64_t pattern_mask[256];
    memset(pattern_mask, 0xFF, sizeof(pattern_mask));
    for (int i = 0; i < m; i++)
        pattern_mask[(unsigned char)target[i]] &= ~(1ULL << i);

    uint64_t R = ~0ULL;              // all bits set = no partial match active
    const uint64_t match_bit = 1ULL << (m - 1);

    for (int i = 0; i < (int)stream.size(); i++) {
        // Shift state left, open a new candidate at bit 0, then kill
        // any position where the current char doesn't match pattern.
        R = (R << 1) | pattern_mask[(unsigned char)stream[i]];

        if (!(R & match_bit))        // bit m-1 went to 0 → full match!
            return i - m + 1;
    }
    return -1;
}

	### 1. High-Level Summary

	This loop scans a "stream" (a string of text) byte by byte, looking for a specific pattern of characters. It does this not by comparing characters one by one in the traditional sense, but by feeding each character into a bitwise state machine.

	Based on the instruction `test rcx, 16` (which checks the 5th bit, $2^4$), we can deduce that **this specific loop is searching for a pattern that is exactly 5 characters long**.

	### 2. Line-by-Line Breakdown

	#### Step 1: Read the character

	```assembly
	movzx  edx, BYTE PTR [rax] 

	```

	* `rax` holds the memory address of the current character in the text stream.
	* `BYTE PTR [rax]` fetches the single byte (character) at that address.
	* `movzx` (Move with Zero-Extend) places that byte into the 32-bit register `edx` and fills the remaining upper bits with zeros. If the character is 'A' (ASCII 65), `edx` now safely holds the integer 65.

	#### Step 2: Update the state

	```assembly
	add    rcx, rcx
	or     rcx, QWORD PTR [rsp+rdx*8]

	```

	* **The Shift:** `rcx` holds the current "state" of the search. `add rcx, rcx` adds the register to itself. In binary arithmetic, adding a number to itself ($x + x = 2x$) is identical to a **left bit-shift by 1** (`state = state << 1`). Compilers often use `add` instead of `shl` (shift left) because it can be slightly more efficient on certain CPU architectures.
	* **The Lookup & OR:** `rsp` is the stack pointer. The code is using the character's ASCII value (`rdx`) as an index into an array of 64-bit (`QWORD`, or 8-byte) masks stored on the stack. `[rsp + rdx*8]` calculates the address of the mask for this specific character. It then performs a bitwise `OR` with the shifted state.

	#### Step 3: Check for the match

	```assembly
	test   rcx, 16
	je     .L_match_found

	```

	* `test rcx, 16` performs a bitwise `AND` between the state (`rcx`) and the number 16 (binary `10000`, which is the 5th bit from the right).
	* In the Shift-Or algorithm, a `0` bit indicates a "live" match. By testing against 16, the code is asking: *"Is the 5th bit of our state a 0?"*

	add    rax, 1
	cmp    rsi, rax
	jne    .L_stream_loop

	1. **The Masks:** Before this loop starts, a lookup table is built. For every possible character (0-255), there is a bitmask. The $i$-th bit of a character's mask is `0` if that character appears at the $i$-th position of the search pattern, and `1` otherwise.
