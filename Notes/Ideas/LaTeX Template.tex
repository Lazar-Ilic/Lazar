values=[1,8,4,6,3,5,2,7]
values.sort()
triplets=[]
for value in values:
    left=0
    right=len(values)-1
    while left<right:
        current=values[left]+values[right]
        if current<value:
            left+=1
        elif current>value:
            right-=1
        else:
            triplets.append([values[left],values[right],value])
            left+=1
            right-=1
print(triplets)
